# TypeScript 基础入门

## 基本类型

### 原始类型

```typescript
// 字符串
let name: string = "张三"

// 数字
let age: number = 25

// 布尔
let isStudent: boolean = true

// null 和 undefined
let n: null = null
let u: undefined = undefined

// Symbol
let sym: symbol = Symbol()

// BigInt
let big: bigint = 100n
```

### 数组

```typescript
// 方式一：类型[]
let numbers: number[] = [1, 2, 3]

// 方式二：Array<类型>
let strings: Array<string> = ["a", "b", "c"]

// 混合类型数组（联合类型）
let mixed: (string | number)[] = [1, "two", 3]
```

### 元组（Tuple）

```typescript
// 固定长度和类型的数组
let tuple: [string, number] = ["张三", 25]

// 可选元素
let optTuple: [string, number?] = ["张三"]

// 只读元组
let readonlyTuple: readonly [string, number] = ["张三", 25]
```

### 枚举（Enum）

```typescript
// 数字枚举
enum Direction {
  Up,      // 0
  Down,    // 1
  Left,    // 2
  Right    // 3
}

// 字符串枚举
enum Status {
  Active = "ACTIVE",
  Inactive = "INACTIVE",
  Pending = "PENDING"
}

// const 枚举（编译时内联，更高效）
const enum Color {
  Red = "RED",
  Green = "GREEN",
  Blue = "BLUE"
}
```

### 特殊类型

```typescript
// any - 任意类型（尽量避免）
let anything: any = "string"
anything = 42

// unknown - 安全的 any
let value: unknown = "hello"
// 必须先检查类型才能使用
if (typeof value === "string") {
  console.log(value.toUpperCase())
}

// void - 函数无返回值
function log(msg: string): void {
  console.log(msg)
}

// never - 永远不会返回
function throwError(msg: string): never {
  throw new Error(msg)
}

// object - 非原始类型
let obj: object = { name: "张三" }
```

## 变量声明

### let 和 const

```typescript
// let - 可变变量
let count = 0
count = 1

// const - 不可变引用
const PI = 3.14
// PI = 3.15  // 错误！
```

### 类型推断

```typescript
// TypeScript 会自动推断类型
let x = 10          // 推断为 number
let y = "hello"     // 推断为 string
let z = true        // 推断为 boolean

// 不需要显式标注
function add(a: number, b: number) {
  return a + b  // 返回类型自动推断为 number
}
```

### 类型断言

```typescript
// 方式一：尖括号语法
let someValue: any = "hello"
let strLength: number = (<string>someValue).length

// 方式二：as 语法（推荐，JSX 兼容）
let strLength2: number = (someValue as string).length

// 非空断言
let element = document.getElementById("app")!
```

## 联合类型与交叉类型

### 联合类型（|）

```typescript
// 可以是多种类型之一
let id: string | number
id = "001"
id = 1

// 函数参数联合类型
function format(value: string | number): string {
  if (typeof value === "string") {
    return value.trim()
  }
  return value.toFixed(2)
}
```

### 交叉类型（&）

```typescript
// 合并多个类型
type Person = {
  name: string
}

type Worker = {
  company: string
}

type Employee = Person & Worker

const emp: Employee = {
  name: "张三",
  company: "ABC 公司"
}
```

## 字面量类型

```typescript
// 字符串字面量
type Direction = "up" | "down" | "left" | "right"
let dir: Direction = "up"

// 数字字面量
type DiceValue = 1 | 2 | 3 | 4 | 5 | 6
let dice: DiceValue = 3

// 模板字面量
type EventName = `on${string}`
let event: EventName = "onClick"
```
