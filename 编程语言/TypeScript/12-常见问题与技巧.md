# 常见问题与技巧

## 类型断言技巧

### 非空断言

```typescript
// 当你确定不是 null/undefined 时
const element = document.getElementById("app")!
const value = maybeNull!
```

### 双重断言（危险，谨慎使用）

```typescript
// 只在必要时使用
let value: string = "hello"
let num = (value as unknown) as number  // 危险！
```

### 类型断言 vs 类型守卫

```typescript
// 类型断言：告诉编译器"相信我"
const value = someValue as string

// 类型守卫：运行时检查
function isString(value: unknown): value is string {
  return typeof value === "string"
}

if (isString(someValue)) {
  // 这里 someValue 自动推断为 string
}
```

## 常见错误

### 对象字面量的额外属性检查

```typescript
interface User {
  name: string
  age: number
}

// 错误：对象字面量会触发额外属性检查
const user: User = {
  name: "张三",
  age: 25,
  email: "test@test.com"  // 错误！
}

// 正确：先赋值给变量
const data = {
  name: "张三",
  age: 25,
  email: "test@test.com"
}
const user: User = data  // 正确！
```

### 类型兼容性

```typescript
interface Animal {
  name: string
}

interface Dog extends Animal {
  breed: string
}

const dog: Dog = { name: "旺财", breed: "金毛" }
const animal: Animal = dog  // 正确！Dog 兼容 Animal

// 反过来不行
// const dog2: Dog = animal  // 错误！Animal 不兼容 Dog
```

### 函数参数的双向协变

```typescript
type Callback = (value: string | number) => void

const fn: Callback = (value: string) => {
  // TypeScript 允许这样，但不安全
  console.log(value.toUpperCase())
}
```

## 类型缩窄

### 穷尽检查

```typescript
type Shape = "circle" | "rectangle" | "triangle"

function getArea(shape: Shape): number {
  switch (shape) {
    case "circle":
      return Math.PI * 10 ** 2
    case "rectangle":
      return 10 * 20
    case "triangle":
      return 0.5 * 10 * 20
    default:
      // 穷尽检查：如果遗漏了某个 case，这里会报错
      const _exhaustive: never = shape
      return _exhaustive
  }
}
```

### 使用 never 进行穷尽检查

```typescript
type Result<T> =
  | { status: "success"; data: T }
  | { status: "error"; message: string }
  | { status: "loading" }

function handleResult<T>(result: Result<T>) {
  switch (result.status) {
    case "success":
      console.log(result.data)
      break
    case "error":
      console.log(result.message)
      break
    case "loading":
      console.log("Loading...")
      break
    default:
      const _exhaustive: never = result
      return _exhaustive
  }
}
```

## 实用工具类型

### Partial 和 Required

```typescript
interface User {
  id: number
  name: string
  email: string
}

// 所有属性可选
type PartialUser = Partial<User>

// 所有属性必选
type RequiredUser = Required<PartialUser>
```

### Pick 和 Omit

```typescript
// 选取部分属性
type UserPreview = Pick<User, "id" | "name">

// 排除部分属性
type UserWithoutEmail = Omit<User, "email">
```

### Record

```typescript
// 构造对象类型
type UserRoles = Record<string, "admin" | "user" | "guest">

const roles: UserRoles = {
  john: "admin",
  jane: "user"
}
```

### Exclude 和 Extract

```typescript
type Status = "active" | "inactive" | "pending"

// 排除
type ActiveStatus = Exclude<Status, "inactive">
// "active" | "pending"

// 提取
type InactiveStatus = Extract<Status, "inactive" | "pending">
// "inactive" | "pending"
```

### NonNullable

```typescript
type NullableString = string | null | undefined
type SafeString = NonNullable<NullableString>
// string
```

### ReturnType 和 Parameters

```typescript
function getUser(id: number) {
  return { id, name: "张三" }
}

type UserReturn = ReturnType<typeof getUser>
// { id: number; name: string }

type GetUserParams = Parameters<typeof getUser>
// [number]
```

## 模式匹配

### infer 关键字

```typescript
// 提取函数返回类型
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never

// 提取数组元素类型
type ArrayElement<T> = T extends (infer E)[] ? E : never

// 提取 Promise 类型
type Awaited<T> = T extends Promise<infer U> ? Awaited<U> : T
```

### 条件类型

```typescript
type IsString<T> = T extends string ? true : false
type IsNumber<T> = T extends number ? true : false

type A = IsString<string>  // true
type B = IsNumber<string>  // false
```

## 类型体操

### 元组转联合

```typescript
type TupleToUnion<T extends any[]> = T[number]

type Result = TupleToUnion<[string, number, boolean]>
// string | number | boolean
```

### 联合转元组

```typescript
type UnionToTuple<T> =
  ((T extends any ? (t: T) => T : never) extends (t: infer U) => U
    ? [U]
    : never) extends [T]
    ? []
    : [...UnionToTuple<Exclude<T, T extends any ? T : never>>]
```

### 深度只读

```typescript
type DeepReadonly<T> = {
  readonly [K in keyof T]: T[K] extends object
    ? T[K] extends Function
      ? T[K]
      : DeepReadonly<T[K]>
    : T[K]
}
```

### 深度 Partial

```typescript
type DeepPartial<T> = {
  [K in keyof T]?: T[K] extends object
    ? T[K] extends Function
      ? T[K]
      : DeepPartial<T[K]>
    : T[K]
}
```

## 性能优化

### 使用 const 断言

```typescript
// 普通枚举
enum Direction {
  Up,      // 0
  Down,    // 1
  Left,    // 2
  Right    // 3
}

// const 枚举（编译时内联）
const enum Direction {
  Up,      // 0
  Down,    // 1
  Left,    // 2
  Right    // 3
}

// const 断言
const direction = "up" as const  // 类型是 "up"，不是 string
```

### 使用 satisfies 操作符

```typescript
type Colors = "red" | "green" | "blue"
type ColorMap = Record<Colors, string>

// 使用 satisfies 保持类型推断
const colors = {
  red: "#ff0000",
  green: "#00ff00",
  blue: "#0000ff"
} satisfies ColorMap

// colors.red 的类型是 string，不是 ColorMap["red"]
// 同时保证了所有属性都存在
```

## 最佳实践

### 1. 使用严格模式

```json
{
  "compilerOptions": {
    "strict": true
  }
}
```

### 2. 避免 any

```typescript
// 不好
function process(data: any) {}

// 好
function process<T>(data: T) {}
```

### 3. 使用 unknown 替代 any

```typescript
// 不好
let value: any = "hello"

// 好
let value: unknown = "hello"
if (typeof value === "string") {
  console.log(value.toUpperCase())
}
```

### 4. 使用类型守卫

```typescript
// 不好
if ((value as any).name) {}

// 好
function hasName(obj: unknown): obj is { name: string } {
  return typeof obj === "object" && obj !== null && "name" in obj
}

if (hasName(value)) {
  console.log(value.name)
}
```

### 5. 使用 as const

```typescript
// 不好
const directions = ["up", "down", "left", "right"]
// 类型是 string[]

// 好
const directions = ["up", "down", "left", "right"] as const
// 类型是 readonly ["up", "down", "left", "right"]
```

### 6. 使用 satisfies

```typescript
// 不好
const config: Config = {
  // 可能丢失具体类型
}

// 好
const config = {
  // 保持具体类型，同时检查是否满足 Config
} satisfies Config
```
