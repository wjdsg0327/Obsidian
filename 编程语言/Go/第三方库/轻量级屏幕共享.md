# 一款用 Go 实现，跨平台支持，无需额外依赖，高效的屏幕实时截图推送服务，适合作为轻量级屏幕共享方案

核心功能：

-  定时捕获主显示器画面（默认10帧/秒）
    
- 将截图编码为 JPEG 格式（可调节质量）
    
- 通过 HTTP 提供实时视频流
    
- 浏览器访问即可查看屏幕直播

```go
package main  
  
import (  
    "bytes"  
    "fmt"    "image"    "image/jpeg"    "log"    "net/http"    "sync"    "sync/atomic" // 用于安全计数  
    "time"  
  
    "github.com/kbinani/screenshot")  
  
var (  
    quality        = 50                     // JPEG质量(1-100)  
    frameRate      = 10                     // 帧率(FPS)  
    screenBounds   = image.Rect(0, 0, 0, 0) // 屏幕尺寸  
    frameCache     []byte  
    frameCacheTime time.Time  
    cacheMutex     sync.RWMutex  
    activeClients  int32 // 当前观看人数计数器  
)  
  
func main() {  
    if n := screenshot.NumActiveDisplays(); n <= 0 {  
       log.Fatal("未检测到活动显示器")  
    }  
    screenBounds = screenshot.GetDisplayBounds(0)  
  
    // 启动后台服务，它会根据 activeClients 的值决定是否真的工作  
    go startCaptureService()  
  
    http.HandleFunc("/stream", streamHandler)  
    http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {  
       // 这里假设你有一个 index.html，如果没有，可以直接输出一个 <img> 标签测试  
       http.ServeFile(w, r, "index.html")  
    })  
  
    log.Println("服务启动: http://localhost:8080")  
    log.Println("提示：无人访问时，后台捕获将进入静默模式以节省CPU。")  
    log.Fatal(http.ListenAndServe(":8080", nil))  
}  
  
func startCaptureService() {  
    ticker := time.NewTicker(time.Second / time.Duration(frameRate))  
    defer ticker.Stop()  
  
    for range ticker.C {  
       // 【关键优化】：如果没有客户端连接，直接跳过，不占 CPU       if atomic.LoadInt32(&activeClients) <= 0 {  
          continue  
       }  
  
       img, err := screenshot.CaptureRect(screenBounds)  
       if err != nil {  
          log.Printf("截屏失败: %v", err)  
          continue  
       }  
  
       buf := new(bytes.Buffer)  
       if err := jpeg.Encode(buf, img, &jpeg.Options{Quality: quality}); err != nil {  
          log.Printf("JPEG编码失败: %v", err)  
          continue  
       }  
  
       updateFrameCache(buf.Bytes())  
    }  
}  
  
func streamHandler(w http.ResponseWriter, r *http.Request) {  
    // 【关键逻辑】：新用户进来，计数器 +1    atomic.AddInt32(&activeClients, 1)  
    log.Printf("新连接加入。当前观看人数: %d", atomic.LoadInt32(&activeClients))  
  
    // 函数退出时（连接断开），计数器 -1    defer func() {  
       atomic.AddInt32(&activeClients, -1)  
       log.Printf("连接断开。当前观看人数: %d", atomic.LoadInt32(&activeClients))  
    }()  
  
    w.Header().Set("Content-Type", "multipart/x-mixed-replace; boundary=frame")  
    w.Header().Set("Cache-Control", "no-cache")  
    w.Header().Set("Connection", "keep-alive")  
  
    if _, err := w.Write([]byte("--frame\r\n")); err != nil {  
       return  
    }  
  
    lastSent := time.Now()  
  
    for {  
       frame, frameTime := getFrameCache()  
  
       // 如果缓存中有新帧，则发送  
       if frameTime.After(lastSent) {  
          if _, err := fmt.Fprintf(w,  
             "Content-Type: image/jpeg\r\nContent-Length: %d\r\n\r\n",  
             len(frame)); err != nil {  
             return  
          }  
          if _, err := w.Write(frame); err != nil {  
             return  
          }  
          if _, err := w.Write([]byte("\r\n--frame\r\n")); err != nil {  
             return  
          }  
          if f, ok := w.(http.Flusher); ok {  
             f.Flush()  
          }  
          lastSent = frameTime  
       }  
  
       // 检查连接是否关闭  
       select {  
       case <-r.Context().Done():  
          return  
       case <-time.After(time.Second / time.Duration(frameRate*2)):  
          // 短暂等待，防止循环空转占满单核CPU  
       }  
    }  
}  
  
func updateFrameCache(frame []byte) {  
    cacheMutex.Lock()  
    defer cacheMutex.Unlock()  
    frameCache = frame  
    frameCacheTime = time.Now()  
}  
  
func getFrameCache() ([]byte, time.Time) {  
    cacheMutex.RLock()  
    defer cacheMutex.RUnlock()  
    return frameCache, frameCacheTime  
}
```
