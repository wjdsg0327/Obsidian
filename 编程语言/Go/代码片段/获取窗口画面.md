//go:build windows  
  
package main  
  
import (  
    "flag"  
    "fmt"    "image/png"    "os"    "strings"    "time")  
  
func main() {  
    title := flag.String("title", "QQ", "要截图的窗口标题（完整标题）")  
    output := flag.String("out", "window.png", "输出 PNG 文件路径")  
    list := flag.Bool("list", false, "列出当前可见窗口")  
    restore := flag.Bool("restore", true, "截图时临时还原最小化窗口")  
    delay := flag.Duration("delay", 500*time.Millisecond, "还原窗口后等待渲染的时间")  
    flag.Parse()  
  
    if *list {  
       windows, err := listVisibleWindows()  
       if err != nil {  
          exitError("枚举窗口失败", err)  
       }  
       for _, window := range windows {  
          if strings.TrimSpace(window.Title) != "" {  
             state := "正常"  
             if window.Minimized {  
                state = "最小化"  
             }  
             fmt.Printf("%d\t%s\t%s\n", window.Handle, state, window.Title)  
          }  
       }  
       return  
    }  
  
    if strings.TrimSpace(*title) == "" {  
       fmt.Fprintln(os.Stderr, "缺少窗口标题；先运行 go run . -list 查看窗口，再使用 -title 指定窗口")  
       flag.Usage()  
       os.Exit(2)  
    }  
  
    hwnd, err := findWindowByTitle(*title)  
    if err != nil {  
       exitError("查找窗口失败", err)  
    }  
  
    img, err := captureWindow(hwnd, *restore, *delay)  
    if err != nil {  
       exitError("截取窗口失败", err)  
    }  
  
    file, err := os.Create(*output)  
    if err != nil {  
       exitError("创建输出文件失败", err)  
    }  
  
    if err := png.Encode(file, img); err != nil {  
       _ = file.Close()  
       exitError("编码 PNG 失败", err)  
    }  
    if err := file.Close(); err != nil {  
       exitError("保存 PNG 失败", err)  
    }  
  
    fmt.Printf("截图成功：%s\n", *output)  
}  
  
func exitError(message string, err error) {  
    fmt.Fprintf(os.Stderr, "%s：%v\n", message, err)  
    os.Exit(1)  
}
