# 08-实战项目：接金币小游戏

## 项目目标

做一个最小可玩的接金币游戏：

- 玩家左右移动
- 金币随机从顶部掉落
- 接到金币加 1 分
- 金币掉到底部扣 1 生命
- 生命为 0 游戏结束
- 可以重新开始

## 场景结构

建议 Hierarchy：

```text
Canvas
├─ GameRoot
│  ├─ Player
│  └─ CoinContainer
├─ UI
│  ├─ ScoreLabel
│  ├─ LifeLabel
│  ├─ StartButton
│  └─ GameOverPanel
└─ GameManager
```

## 第一步：Player

创建 Player 节点：

- 用 Sprite 或简单色块显示
- 加 PlayerController 脚本
- 能左右移动
- 限制不出屏幕

`PlayerController.ts`：

```ts
import { _decorator, Component, input, Input, EventKeyboard, KeyCode } from 'cc';
const { ccclass, property } = _decorator;

@ccclass('PlayerController')
export class PlayerController extends Component {
  @property
  speed = 400;

  @property
  minX = -300;

  @property
  maxX = 300;

  private dir = 0;

  onLoad() {
    input.on(Input.EventType.KEY_DOWN, this.onKeyDown, this);
    input.on(Input.EventType.KEY_UP, this.onKeyUp, this);
  }

  onDestroy() {
    input.off(Input.EventType.KEY_DOWN, this.onKeyDown, this);
    input.off(Input.EventType.KEY_UP, this.onKeyUp, this);
  }

  onKeyDown(event: EventKeyboard) {
    if (event.keyCode === KeyCode.ARROW_LEFT || event.keyCode === KeyCode.KEY_A) this.dir = -1;
    if (event.keyCode === KeyCode.ARROW_RIGHT || event.keyCode === KeyCode.KEY_D) this.dir = 1;
  }

  onKeyUp(event: EventKeyboard) {
    if ((event.keyCode === KeyCode.ARROW_LEFT || event.keyCode === KeyCode.KEY_A) && this.dir === -1) this.dir = 0;
    if ((event.keyCode === KeyCode.ARROW_RIGHT || event.keyCode === KeyCode.KEY_D) && this.dir === 1) this.dir = 0;
  }

  update(dt: number) {
    const pos = this.node.position.clone();
    pos.x += this.dir * this.speed * dt;
    pos.x = Math.max(this.minX, Math.min(this.maxX, pos.x));
    this.node.setPosition(pos);
  }
}
```

## 第二步：Coin 预制体

创建 Coin 节点，做成 Prefab：

- Sprite 显示金币
- BoxCollider2D 或 CircleCollider2D
- Coin.ts 控制下落

`Coin.ts`：

```ts
import { _decorator, Component } from 'cc';
const { ccclass, property } = _decorator;

@ccclass('Coin')
export class Coin extends Component {
  @property
  fallSpeed = 250;

  private gameManager: any = null;
  private collected = false;

  init(gameManager: any) {
    this.gameManager = gameManager;
  }

  update(dt: number) {
    if (this.collected) return;

    const pos = this.node.position.clone();
    pos.y -= this.fallSpeed * dt;
    this.node.setPosition(pos);

    if (pos.y < -420) {
      if (this.gameManager) this.gameManager.missCoin();
      this.node.destroy();
    }
  }

  collect() {
    if (this.collected) return;
    this.collected = true;
    if (this.gameManager) this.gameManager.addScore(1);
    this.node.destroy();
  }
}
```

> 为了教程简单，这里先用 `any`，正式项目可以用明确类型。

## 第三步：GameManager

`GameManager.ts`：

```ts
import { _decorator, Component, Label, Node, Prefab, instantiate, randomRange } from 'cc';
import { Coin } from './Coin';
const { ccclass, property } = _decorator;

@ccclass('GameManager')
export class GameManager extends Component {
  @property(Label)
  scoreLabel: Label | null = null;

  @property(Label)
  lifeLabel: Label | null = null;

  @property(Node)
  startButton: Node | null = null;

  @property(Node)
  gameOverPanel: Node | null = null;

  @property(Prefab)
  coinPrefab: Prefab | null = null;

  @property(Node)
  coinContainer: Node | null = null;

  score = 0;
  life = 3;
  playing = false;
  spawnTimer = 0;

  start() {
    this.resetUI();
  }

  public startGame() {
    this.score = 0;
    this.life = 3;
    this.playing = true;
    this.spawnTimer = 0;
    if (this.startButton) this.startButton.active = false;
    if (this.gameOverPanel) this.gameOverPanel.active = false;
    this.clearCoins();
    this.refreshUI();
  }

  update(dt: number) {
    if (!this.playing) return;

    this.spawnTimer += dt;
    if (this.spawnTimer >= 1) {
      this.spawnTimer = 0;
      this.spawnCoin();
    }
  }

  spawnCoin() {
    if (!this.coinPrefab || !this.coinContainer) return;

    const coinNode = instantiate(this.coinPrefab);
    this.coinContainer.addChild(coinNode);
    coinNode.setPosition(randomRange(-300, 300), 360, 0);

    const coin = coinNode.getComponent(Coin);
    if (coin) coin.init(this);
  }

  addScore(value: number) {
    if (!this.playing) return;
    this.score += value;
    this.refreshUI();
  }

  missCoin() {
    if (!this.playing) return;
    this.life -= 1;
    this.refreshUI();
    if (this.life <= 0) {
      this.gameOver();
    }
  }

  gameOver() {
    this.playing = false;
    if (this.gameOverPanel) this.gameOverPanel.active = true;
    if (this.startButton) this.startButton.active = true;
  }

  clearCoins() {
    if (!this.coinContainer) return;
    this.coinContainer.removeAllChildren();
  }

  resetUI() {
    if (this.gameOverPanel) this.gameOverPanel.active = false;
    this.refreshUI();
  }

  refreshUI() {
    if (this.scoreLabel) this.scoreLabel.string = `分数：${this.score}`;
    if (this.lifeLabel) this.lifeLabel.string = `生命：${this.life}`;
  }
}
```

## 第四步：接金币判断

最简单的入门方案：用距离判断，不先折腾物理碰撞。

在 GameManager 里每帧检查 Player 和 Coin 距离也可以。但更推荐后续用 Collider2D。

如果你先想快速跑通游戏：

- 给 GameManager 增加 player 引用
- 遍历 coinContainer children
- 如果 x/y 距离足够近，就调用 coin.collect()

伪代码：

```ts
for (const coinNode of this.coinContainer.children) {
  const dx = Math.abs(coinNode.position.x - this.player.position.x);
  const dy = Math.abs(coinNode.position.y - this.player.position.y);
  if (dx < 60 && dy < 60) {
    coinNode.getComponent(Coin)?.collect();
  }
}
```

这不是最“引擎标准”的方法，但非常适合第一版原型。
