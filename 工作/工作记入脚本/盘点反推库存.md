package main  
  
import (  
    "context"  
    "database/sql"    "errors"    "flag"    "fmt"    "log"    "net"    "os"    "strings"    "time"  
    "github.com/go-sql-driver/mysql"    "gopkg.in/yaml.v3")  
  
const defaultTimeout = 30 * time.Minute  
  
type Config struct {  
    MySQL MySQLConfig `yaml:"mysql"`  
}  
  
type MySQLConfig struct {  
    Host            string `yaml:"host"`  
    Port            int    `yaml:"port"`  
    User            string `yaml:"user"`  
    Password        string `yaml:"password"`  
    Database        string `yaml:"database"`  
    Charset         string `yaml:"charset"`  
    MaxOpenConns    int    `yaml:"max_open_conns"`  
    MaxIdleConns    int    `yaml:"max_idle_conns"`  
    ConnMaxLifetime string `yaml:"conn_max_lifetime"`  
}  
  
type productJob struct {  
    Name        string  
    Table       string  
    SelectQuery string  
}  
  
type candidate struct {  
    ID            string  
    PositionName  sql.NullString  
    ShelfName     sql.NullString  
    BelongingZone sql.NullString  
    WarehouseName sql.NullString  
}  
  
type jobStats struct {  
    SourceRows       int  
    Matched          int  
    Unmatched        int  
    Skipped          int  
    Updated          int64  
    UnmatchedDetails []detailInfo  
    SkippedDetails   []detailInfo  
}  
  
type detailInfo struct {  
    DetailID    string  
    OddNumber   sql.NullString  
    ProductCode sql.NullString  
    ProductSpec sql.NullString  
    ProductName sql.NullString  
    Position    sql.NullString  
    SkipReason  string  
}  
  
type queryer interface {  
    QueryContext(context.Context, string, ...any) (*sql.Rows, error)  
}

func candidateQuery(documentType, table, productJoin string) string {  
    return fmt.Sprintf(`  
SELECT  
    d.id AS detail_id,     t.oddNumber,    d.product_code,    d.product_spc,     d.product_name,     t.presets1,    w.id AS location_id,    p.id AS product_id,    w.position_name,    w.shelf_name,    w.belonging_zone,    w.warehouse_ameFROM zhyh_stock_take AS t  
JOIN zhyh_stock_take_detail AS d  
  ON t.oddNumber = d.mainOddLEFT JOIN zhyh_warehouse_location_management AS w  
  ON t.presets1 = w.position_nameLEFT JOIN %s AS p  
  ON %sWHERE t.document_type = '%s'  
ORDER BY  
    (t.created_at IS NULL) ASC,    t.created_at DESC,    t.id DESC,    d.id DESC,    p.id ASC`, table, productJoin, documentType)  
}  
  
func main() {  
    configPath := flag.String("config", "config.yaml", "YAML configuration file")  
    apply := flag.Bool("apply", false, "commit updates; without this flag the program only previews")  
    timeout := flag.Duration("timeout", defaultTimeout, "maximum runtime, for example 10m or 1h")  
    flag.Parse()  
  
    if err := run(*configPath, *apply, *timeout); err != nil {  
       log.Printf("执行失败: %v", err)  
       os.Exit(1)  
    }  
}  
  
func run(configPath string, apply bool, timeout time.Duration) error {  
    if timeout <= 0 {  
       return errors.New("timeout 必须大于 0")  
    }  
  
    cfg, err := loadConfig(configPath)  
    if err != nil {  
       return err  
    }  
  
    db, err := openDatabase(cfg.MySQL)  
    if err != nil {  
       return err  
    }  
    defer db.Close()  
  
    ctx, cancel := context.WithTimeout(context.Background(), timeout)  
    defer cancel()  
  
    if err := db.PingContext(ctx); err != nil {  
       return fmt.Errorf("连接 MySQL 失败: %w", err)  
    }  
  
    mode := "预览"  
    if apply {  
       mode = "更新"  
    }  
    log.Printf("已连接 MySQL %s:%d/%s，运行模式: %s", cfg.MySQL.Host, cfg.MySQL.Port, cfg.MySQL.Database, mode)  
  
    if !apply {  
       return preview(ctx, db)  
    }  
    return applyUpdates(ctx, db)  
}  
  
func loadConfig(path string) (Config, error) {  
    data, err := os.ReadFile(path)  
    if err != nil {  
       return Config{}, fmt.Errorf("读取配置文件 %q 失败: %w", path, err)  
    }  
  
    var cfg Config  
    if err := yaml.Unmarshal(data, &cfg); err != nil {  
       return Config{}, fmt.Errorf("解析配置文件 %q 失败: %w", path, err)  
    }  
  
    if cfg.MySQL.Host == "" || cfg.MySQL.User == "" || cfg.MySQL.Database == "" {  
       return Config{}, errors.New("mysql.host、mysql.user 和 mysql.database 不能为空")  
    }  
    if cfg.MySQL.Port == 0 {  
       cfg.MySQL.Port = 3306  
    }  
    if cfg.MySQL.Port < 1 || cfg.MySQL.Port > 65535 {  
       return Config{}, fmt.Errorf("mysql.port 无效: %d", cfg.MySQL.Port)  
    }  
    if cfg.MySQL.Charset == "" {  
       cfg.MySQL.Charset = "utf8mb4"  
    }  
    if cfg.MySQL.MaxOpenConns <= 0 {  
       cfg.MySQL.MaxOpenConns = 5  
    }  
    if cfg.MySQL.MaxIdleConns < 0 {  
       return Config{}, errors.New("mysql.max_idle_conns 不能小于 0")  
    }  
    if cfg.MySQL.MaxIdleConns == 0 {  
       cfg.MySQL.MaxIdleConns = 2  
    }  
    if cfg.MySQL.ConnMaxLifetime == "" {  
       cfg.MySQL.ConnMaxLifetime = "30m"  
    }  
    if _, err := time.ParseDuration(cfg.MySQL.ConnMaxLifetime); err != nil {  
       return Config{}, fmt.Errorf("mysql.conn_max_lifetime 无效: %w", err)  
    }  
  
    return cfg, nil  
}  
  
func openDatabase(cfg MySQLConfig) (*sql.DB, error) {  
    lifetime, _ := time.ParseDuration(cfg.ConnMaxLifetime)  
    driverConfig := mysql.NewConfig()  
    driverConfig.User = cfg.User  
    driverConfig.Passwd = cfg.Password  
    driverConfig.Net = "tcp"  
    driverConfig.Addr = net.JoinHostPort(cfg.Host, fmt.Sprintf("%d", cfg.Port))  
    driverConfig.DBName = cfg.Database  
    driverConfig.Collation = cfg.Charset + "_0900_ai_ci"  
    driverConfig.ParseTime = true  
    driverConfig.Loc = time.Local  
    driverConfig.Timeout = 10 * time.Second  
    driverConfig.ReadTimeout = 0  
    driverConfig.WriteTimeout = 0  
  
    db, err := sql.Open("mysql", driverConfig.FormatDSN())  
    if err != nil {  
       return nil, fmt.Errorf("初始化 MySQL 连接失败: %w", err)  
    }  
    db.SetMaxOpenConns(cfg.MaxOpenConns)  
    db.SetMaxIdleConns(cfg.MaxIdleConns)  
    db.SetConnMaxLifetime(lifetime)  
    return db, nil  
}  
  
func preview(ctx context.Context, db *sql.DB) error {  
    for _, job := range jobs {  
       _, stats, err := collectCandidates(ctx, db, job)  
       if err != nil {  
          return fmt.Errorf("预览 %s 失败: %w", job.Name, err)  
       }  
       printStats(job, stats, false)  
    }  
    log.Print("预览完成，数据库未发生修改；确认结果后添加 -apply 参数执行更新")  
    return nil  
}  
  
func applyUpdates(ctx context.Context, db *sql.DB) error {  
    tx, err := db.BeginTx(ctx, &sql.TxOptions{Isolation: sql.LevelRepeatableRead})  
    if err != nil {  
       return fmt.Errorf("开启事务失败: %w", err)  
    }  
    defer tx.Rollback()  
  
    results := make([]jobStats, 0, len(jobs))  
    for _, job := range jobs {  
       candidates, stats, err := collectCandidates(ctx, tx, job)  
       if err != nil {  
          return fmt.Errorf("查询 %s 更新候选失败: %w", job.Name, err)  
       }  
       updated, err := updateCandidates(ctx, tx, job, candidates)  
       if err != nil {  
          return fmt.Errorf("更新 %s 失败，事务将回滚: %w", job.Name, err)  
       }  
       stats.Updated = updated  
       results = append(results, stats)  
    }  
  
    if err := tx.Commit(); err != nil {  
       return fmt.Errorf("提交事务失败: %w", err)  
    }  
    for i, job := range jobs {  
       printStats(job, results[i], true)  
    }  
    log.Print("所有更新已在同一事务中提交")  
    return nil  
}  
  
func collectCandidates(ctx context.Context, q queryer, job productJob) ([]candidate, jobStats, error) {  
    rows, err := q.QueryContext(ctx, job.SelectQuery)  
    if err != nil {  
       return nil, jobStats{}, err  
    }  
    defer rows.Close()  
  
    var stats jobStats  
    seenDetails := make(map[string]struct{})  
    detailOrder := make([]string, 0)  
    detailByID := make(map[string]detailInfo)  
    skippedDetails := make(map[string]struct{})  
    matchedDetails := make(map[string]struct{})  
    seenProducts := make(map[string]struct{})  
    candidates := make([]candidate, 0)  
  
    for rows.Next() {  
       var detail detailInfo  
       var locationID, productID sql.NullString  
       var item candidate  
       if err := rows.Scan(  
          &detail.DetailID,  
          &detail.OddNumber,  
          &detail.ProductCode,  
          &detail.ProductSpec,  
          &detail.ProductName,  
          &detail.Position,  
          &locationID,  
          &productID,  
          &item.PositionName,  
          &item.ShelfName,  
          &item.BelongingZone,  
          &item.WarehouseName,  
       ); err != nil {  
          return nil, stats, err  
       }  
  
       _, detailSeen := seenDetails[detail.DetailID]  
       if !detailSeen {  
          seenDetails[detail.DetailID] = struct{}{}  
          detailOrder = append(detailOrder, detail.DetailID)  
          detailByID[detail.DetailID] = detail  
          stats.SourceRows++  
       }  
  
       missingSpec := job.Name != "物料商品" && blank(detail.ProductSpec)  
       if blank(detail.ProductCode) || missingSpec || !locationID.Valid {  
          if !detailSeen {  
             reasons := make([]string, 0, 3)  
             if blank(detail.ProductCode) {  
                reasons = append(reasons, "商品编码为空")  
             }  
             if missingSpec {  
                reasons = append(reasons, "商品规格为空")  
             }  
             if !locationID.Valid {  
                reasons = append(reasons, "仓位表找不到该仓位")  
             }  
             detail.SkipReason = strings.Join(reasons, "；")  
             detailByID[detail.DetailID] = detail  
             stats.Skipped++  
             skippedDetails[detail.DetailID] = struct{}{}  
          }  
          continue  
       }  
       if !productID.Valid {  
          continue  
       }  
  
       matchedDetails[detail.DetailID] = struct{}{}  
       if _, exists := seenProducts[productID.String]; exists {  
          continue  
       }  
       seenProducts[productID.String] = struct{}{}  
       item.ID = productID.String  
       candidates = append(candidates, item)  
    }  
    if err := rows.Err(); err != nil {  
       return nil, stats, err  
    }  
  
    stats.Matched = len(candidates)  
    for _, detailID := range detailOrder {  
       if _, skipped := skippedDetails[detailID]; skipped {  
          stats.SkippedDetails = append(stats.SkippedDetails, detailByID[detailID])  
          continue  
       }  
       if _, matched := matchedDetails[detailID]; matched {  
          continue  
       }  
       stats.UnmatchedDetails = append(stats.UnmatchedDetails, detailByID[detailID])  
    }  
    stats.Unmatched = len(stats.UnmatchedDetails)  
    return candidates, stats, nil  
}  
  
func blank(value sql.NullString) bool {  
    if !value.Valid {  
       return true  
    }  
    for _, r := range value.String {  
       if r != ' ' && r != '\t' && r != '\r' && r != '\n' {  
          return false  
       }  
    }  
    return true  
}  
  
func updateCandidates(ctx context.Context, tx *sql.Tx, job productJob, candidates []candidate) (int64, error) {  
    query := fmt.Sprintf(`  
UPDATE %s  
SET position_name = ?,  
    shelf_name = ?,    belonging_zone = ?,    warehouse_ame = ?WHERE id = ?`, job.Table)  
    stmt, err := tx.PrepareContext(ctx, query)  
    if err != nil {  
       return 0, err  
    }  
    defer stmt.Close()  
  
    var updated int64  
    for _, item := range candidates {  
       result, err := stmt.ExecContext(  
          ctx,  
          item.PositionName,  
          item.ShelfName,  
          item.BelongingZone,  
          item.WarehouseName,  
          item.ID,  
       )  
       if err != nil {  
          return updated, fmt.Errorf("商品 id=%s: %w", item.ID, err)  
       }  
       rowsAffected, err := result.RowsAffected()  
       if err != nil {  
          return updated, fmt.Errorf("读取商品 id=%s 更新结果失败: %w", item.ID, err)  
       }  
       updated += rowsAffected  
    }  
    return updated, nil  
}  
  
func printStats(job productJob, stats jobStats, applied bool) {  
    if applied {  
       log.Printf("[%s -> %s] 明细=%d，匹配商品=%d，实际变更=%d，未匹配明细=%d，跳过明细=%d",  
          job.Name, job.Table, stats.SourceRows, stats.Matched, stats.Updated, stats.Unmatched, stats.Skipped)  
       printUnmatchedDetails(job, stats.UnmatchedDetails)  
       printSkippedDetails(job, stats.SkippedDetails)  
       return  
    }  
    log.Printf("[%s -> %s] 明细=%d，预计更新商品=%d，未匹配明细=%d，跳过明细=%d",  
       job.Name, job.Table, stats.SourceRows, stats.Matched, stats.Unmatched, stats.Skipped)  
    printUnmatchedDetails(job, stats.UnmatchedDetails)  
    printSkippedDetails(job, stats.SkippedDetails)  
}  
  
func printUnmatchedDetails(job productJob, details []detailInfo) {  
    for _, detail := range details {  
       log.Printf("[%s 未匹配] 盘点单=%s，明细ID=%s，商品编码=%s，规格=%s，商品名称=%s，仓位=%s",  
          job.Name,  
          displayNullString(detail.OddNumber),  
          detail.DetailID,  
          displayNullString(detail.ProductCode),  
          displayNullString(detail.ProductSpec),  
          displayNullString(detail.ProductName),  
          displayNullString(detail.Position),  
       )  
    }  
}  
  
func printSkippedDetails(job productJob, details []detailInfo) {  
    for _, detail := range details {  
       log.Printf("[%s 跳过] 原因=%s，盘点单=%s，明细ID=%s，商品编码=%s，规格=%s，商品名称=%s，仓位=%s",  
          job.Name,  
          detail.SkipReason,  
          displayNullString(detail.OddNumber),  
          detail.DetailID,  
          displayNullString(detail.ProductCode),  
          displayNullString(detail.ProductSpec),  
          displayNullString(detail.ProductName),  
          displayNullString(detail.Position),  
       )  
    }  
}  
  
func displayNullString(value sql.NullString) string {  
    if !value.Valid || value.String == "" {  
       return "<空>"  
    }  
    return value.String  
}
