#### Go 语言方案

**优势：**
- 高并发性能优异（Goroutine）
- 编译型语言，部署简单
- 丰富的云原生生态
- Milvus、Weaviate 等向量数据库用 Go 编写

**技术栈：**
```

向量数据库客户端：Milvus Go SDK、Qdrant Go SDK
LLM 客户端：go-openai、langchaingo
文档处理：unidoc/unipdf、go-docx
消息队列：Kafka Go SDK、NATS
缓存：go-redis
```

**适合场景：**
- 高并发查询场景
- 微服务架构
- 云原生部署（K8s）

**示例代码（Go + Milvus）：**
```go
package main

import (
    "context"
    "log"

    "github.com/milvus-io/milvus-sdk-go/v2/client"
    "github.com/milvus-io/milvus-sdk-go/v2/entity"
)

func main() {
    // 连接 Milvus
    ctx := context.Background()
    c, err := client.NewGrpcClient(ctx, "localhost:19530")
    if err != nil {
        log.Fatal("Failed to connect to Milvus:", err)
    }
    defer c.Close()

    // 创建集合（知识库表）
    schema := entity.NewSchema().
        WithField(entity.NewField().
            WithName("id").
            WithDataType(entity.FieldType_Int64).
            WithIsPrimaryKey(true)).
        WithField(entity.NewField().
            WithName("content").
            WithDataType(entity.FieldType_VarChar).
            WithMaxLength(65535)).
        WithField(entity.NewField().
            WithName("embedding").
            WithDataType(entity.FieldType_FloatVector).
            WithDim(1536)) // OpenAI embedding 维度

    err = c.CreateCollection(ctx, schema, entity.DefaultShardNumber)
    if err != nil {
        log.Fatal("Failed to create collection:", err)
    }

    // 创建索引（HNSW）
    idx, err := entity.NewIndexHNSW(entity.L2, 16, 200)
    if err != nil {
        log.Fatal("Failed to create index:", err)
    }
    err = c.CreateIndex(ctx, "knowledge_base", "embedding", idx, false)
    if err != nil {
        log.Fatal("Failed to create index:", err)
    }

    // 插入数据
    ids := entity.NewColumnInt64("id", []int64{1, 2, 3})
    contents := entity.NewColumnVarChar("content", []string{
        "企业知识库是存储企业内部文档的系统",
        "RAG 是检索增强生成的缩写",
        "向量数据库用于存储和检索 embedding 向量",
    })
    embeddings := entity.NewColumnFloatVector("embedding", 1536, [][]float32{
        /* 这里应该是实际的 embedding 向量 */
    })

    _, err = c.Insert(ctx, "knowledge_base", "", ids, contents, embeddings)
    if err != nil {
        log.Fatal("Failed to insert data:", err)
    }

    // 加载集合到内存
    err = c.LoadCollection(ctx, "knowledge_base", false)
    if err != nil {
        log.Fatal("Failed to load collection:", err)
    }

    // 检索相似文档
    searchVectors := [][]float32{
        /* 查询向量 */
    }
    results, err := c.Search(ctx, "knowledge_base", []string{},
        []string{"content"}, // 输出字段
        "",
        []entity.Vector{
            entity.FloatVector(searchVectors[0]),
        },
        "embedding",
        entity.L2,
        10, // top_k
        entity.WithSearchParam("ef", "200"),
    )
    if err != nil {
        log.Fatal("Failed to search:", err)
    }

    // 处理结果
    for _, result := range results {
        for _, id := range result.IDs.(*entity.ColumnInt64).Data() {
            log.Printf("Found document ID: %d", id)
        }
    }
}
```

import dev.langchain4j.data.document.Document;
import dev.langchain4j.data.document.loader.FileSystemDocumentLoader;
import dev.langchain4j.data.embedding.Embedding;
import dev.langchain4j.data.segment.TextSegment;
import dev.langchain4j.model.openai.OpenAiEmbeddingModel;
import dev.langchain4j.store.embedding.EmbeddingStore;
import dev.langchain4j.store.embedding.milvus.MilvusEmbeddingStore;

import java.util.List;

public class KnowledgeBaseService {

    private final EmbeddingStore<TextSegment> embeddingStore;
    private final OpenAiEmbeddingModel embeddingModel;

    public KnowledgeBaseService() {
        // 初始化 Milvus 向量存储
        this.embeddingStore = MilvusEmbeddingStore.builder()
            .host("localhost")
            .port(19530)
            .collectionName("knowledge_base")
            .dimension(1536)
            .build();

        // 初始化 Embedding 模型
        this.embeddingModel = OpenAiEmbeddingModel.builder()
            .apiKey(System.getenv("OPENAI_API_KEY"))
            .modelName("text-embedding-3-large")
            .build();
    }

    /**
     * 文档入库
     */
    public void ingestDocument(String filePath) {
        // 加载文档
        List<Document> documents = FileSystemDocumentLoader
            .loadDocumentsRecursively(filePath);

        for (Document doc : documents) {
            // 分块（256 tokens，50 tokens 重叠）
            List<TextSegment> chunks = splitDocument(doc, 256, 50);

            for (TextSegment chunk : chunks) {
                // 向量化
                Embedding embedding = embeddingModel.embed(chunk).content();

                // 存储到 Milvus
                embeddingStore.add(embedding, chunk);
            }
        }
    }

    /**
     * 查询知识库
     */
    public String query(String question) {
        // 1. 问题向量化
        Embedding queryEmbedding = embeddingModel.embed(question).content();

        // 2. 向量检索 Top-K
        List<EmbeddingMatch<TextSegment>> matches = 
            embeddingStore.findRelevant(queryEmbedding, 5);

        // 3. 构建上下文
        StringBuilder context = new StringBuilder();
        for (EmbeddingMatch<TextSegment> match : matches) {
            context.append(match.embedded().text()).append("\n\n");

            基于以下参考资料回答问题。如果资料中没有相关信息，请说明。

            %s
            
            问题：%s
            
            请给出准确、详细的答案，并引用来源。
            """, context, question);

        return callLLM(prompt);

### 3.2 分块策略

| 策略 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| **固定大小** | 简单快速 | 破坏语义 | 快速原型 |
| **递归字符** | 尊重边界 | 需要调参 | 通用文档 |
| **语义分块** | 质量最高 | 计算成本高 | 高质量需求 |
| **文档结构** | 保留层次 | 依赖格式 | 结构化文档 |
| **Agentic** | 智能最优 | 最慢 | 高价值内容 |

**推荐配置：**
```
chunk_size: 512 tokens
chunk_overlap: 50-100 tokens (10-20%)
separator: ["\n\n", "\n", "。", "！", "？", ".", "!", "?"]
```

### 3.3 高级 RAG 技术

#### 预检索优化
- **查询改写**：将模糊查询转为精确查询
- **HyDE**：先生成假设答案，再用假设答案检索
- **Step-back Prompting**：先抽象问题再检索
- **子查询分解**：将复杂问题拆分为多个子问题

#### 检索优化
- **混合检索**：向量检索 + BM25 关键词检索
- **元数据过滤**：按时间、来源、类别过滤
- **多跳检索**：迭代检索复杂推理
- **Parent-Child 分块**：检索小块，返回大块上下文

#### 后检索优化
- **重排序**：使用 Cross-Encoder 模型重排
- **上下文压缩**：只提取相关内容
- **Lost-in-the-Middle**：将相关内容放在开头
