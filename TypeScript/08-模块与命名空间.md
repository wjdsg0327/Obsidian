# 模块与命名空间

## ES 模块

### 导出（Export）

```typescript
// 命名导出
export const name = "张三"
export const age = 25

export function greet(name: string): string {
  return `Hello, ${name}`
}

export interface User {
  id: number
  name: string
}

// 默认导出
export default class UserService {
  getUser(id: number): User {
    return { id, name: "张三" }
  }
}
```

### 导入（Import）

```typescript
// 导入命名导出
import { name, age, greet } from "./user"
import { User } from "./user"

// 导入默认导出
import UserService from "./user"

// 导入全部
import * as userModule from "./user"
userModule.greet("张三")

// 重命名导入
import { name as userName } from "./user"

// 混合导入
import UserService, { name, age } from "./user"
```

### 重新导出（Re-export）

```typescript
// 重新导出命名
export { name, age } from "./user"

// 重新导出全部
export * from "./user"

// 重命名重新导出
export { name as userName } from "./user"

// 重新导出默认
export { default as UserService } from "./user"
```

## 动态导入

### import()

```typescript
// 动态导入
const module = await import("./module")
module.doSomething()

// 按需加载
async function loadFeature() {
  const { Feature } = await import("./feature")
  return new Feature()
}

// 条件导入
if (process.env.NODE_ENV === "development") {
  const { DevTools } = await import("./dev-tools")
  DevTools.init()
}
```

## 类型导入

### import type

```typescript
// 只导入类型，不导入值
import type { User, Config } from "./types"

// 内联类型导入
import { type User, createUser } from "./user"

// 导入默认类型
import type DefaultType from "./module"
```

### 为什么使用 import type

```typescript
// 编译后会被完全移除
import type { User } from "./user"

// 编译后会保留
import { User } from "./user"

// 好处：
// 1. 明确表示只用于类型检查
// 2. 避免循环依赖问题
// 3. 减少打包体积
```

## 命名空间（Namespace）

### 基础命名空间

```typescript
namespace Validation {
  export interface Validator {
    validate(value: string): boolean
  }

  export class EmailValidator implements Validator {
    validate(value: string): boolean {
      return value.includes("@")
    }
  }

  export class PhoneValidator implements Validator {
    validate(value: string): boolean {
      return /^\d{11}$/.test(value)
    }
  }
}

// 使用
const emailValidator = new Validation.EmailValidator()
emailValidator.validate("test@example.com")
```

### 命名空间拆分

```typescript
// validation.ts
namespace Validation {
  export interface Validator {
    validate(value: string): boolean
  }
}

// email-validator.ts
/// <reference path="validation.ts" />
namespace Validation {
  export class EmailValidator implements Validator {
    validate(value: string): boolean {
      return value.includes("@")
    }
  }
}

// 使用时需要引入所有文件
```

### 命名空间与模块

```typescript
// 不推荐：使用命名空间
namespace MyLib {
  export function doSomething() {}
}

// 推荐：使用模块
export function doSomething() {}
```

## 模块解析

### 相对路径

```typescript
// 同目录
import { user } from "./user"

// 上级目录
import { config } from "../config"

// 子目录
import { helper } from "./utils/helper"
```

### 非相对路径（模块路径）

```typescript
// 从 node_modules 导入
import express from "express"
import React from "react"

// 路径别名（需要配置）
import { User } from "@models/user"
import { api } from "@services/api"
```

### 路径别名配置

```json
// tsconfig.json
{
  "compilerOptions": {
    "baseUrl": ".",
    "paths": {
      "@/*": ["src/*"],
      "@models/*": ["src/models/*"],
      "@services/*": ["src/services/*"]
    }
  }
}
```

## 模块导出模式

### 导出聚合（Barrel）

```typescript
// models/index.ts
export { User } from "./user"
export { Post } from "./post"
export { Comment } from "./comment"

// 使用
import { User, Post, Comment } from "./models"
```

### 命名导出 vs 默认导出

```typescript
// 命名导出（推荐）
export function createUser() {}
export class UserService {}
export interface User {}

// 默认导出（单个主要导出时使用）
export default class App {
  constructor() {}
}

// 混合使用
export { UserService, User }
export default App
```

### 条件导出

```typescript
// utils.ts
export function isServer(): boolean {
  return typeof window === "undefined"
}

export function isClient(): boolean {
  return typeof window !== "undefined"
}

// 服务器专用
export const serverOnly = isServer()
  ? require("./server-utils")
  : null
```

## CommonJS 与 ES 模块互操作

### 在 TypeScript 中使用 CommonJS

```typescript
// 导入 CommonJS 模块
import express = require("express")

// 或者使用 ES 模块语法（推荐）
import express from "express"
```

### 导出为 CommonJS

```typescript
// TypeScript 自动处理
export function doSomething() {}
export default class MyClass {}

// 编译后
// exports.doSomething = function() {}
// exports.default = MyClass
```

### 混合模块系统

```typescript
// types.d.ts
declare module "legacy-lib" {
  export function oldFunction(): void
}

// 使用
import { oldFunction } from "legacy-lib"
```

## 实战示例

### API 模块组织

```typescript
// api/index.ts
export { userApi } from "./user"
export { postApi } from "./post"
export { commentApi } from "./comment"

// api/user.ts
import { request } from "./request"

export const userApi = {
  getUser(id: number) {
    return request.get<User>(`/users/${id}`)
  },
  createUser(data: CreateUserDto) {
    return request.post<User>("/users", data)
  }
}

// api/post.ts
export const postApi = {
  getPosts() {
    return request.get<Post[]>("/posts")
  }
}
```

### 工具模块

```typescript
// utils/index.ts
export * from "./format"
export * from "./validate"
export * from "./storage"

// utils/format.ts
export function formatDate(date: Date): string {
  return date.toLocaleDateString("zh-CN")
}

export function formatMoney(amount: number): string {
  return `¥${amount.toFixed(2)}`
}

// utils/validate.ts
export function isEmail(value: string): boolean {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)
}

export function isPhone(value: string): boolean {
  return /^1\d{10}$/.test(value)
}
```

### 类型定义模块

```typescript
// types/index.ts
export * from "./user"
export * from "./api"
export * from "./common"

// types/user.ts
export interface User {
  id: number

  name: string
  email: string
  password: string

// types/api.ts

  code: number
  message: string
  data: T
}

export interface PaginatedResponse<T> {
  items: T[]
  total: number
  page: number
  pageSize: number
}
```
